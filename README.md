Example Required Environment Variables:

	CONSOLE_LOG_LEVEL = "info"
	CATAPULT_API_PORT = 5050
	CATAPULT_API_HOST = "localhost"
	CATAPULT_USER_ID = "u-user1"
	CATAPULT_API_TOKEN = "t-token1"
	CATAPULT_API_SECRET = "secret"
